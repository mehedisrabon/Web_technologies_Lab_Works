function capitalize() {
    var x = document.getElementById("input1").value;
    var p = x.split(" ");
    var y = "";
    
    for (var i = 0; i < p.length; i++) {
        y += p[i].charAt(0).toUpperCase() + p[i].slice(1) + " ";
    }

    document.getElementById("output1").innerHTML = y.trim();
}
